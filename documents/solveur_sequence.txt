    activate solver #FFBBBB
    solver -> solver : evaluate_attempt_score(w, …)
    activate solver #DarkSalmon
    loop <i>Pour tous les mots "c1"</i><i>possiblement valides avec les indices actuels</i>
        note over solver
            On part du principe que c1 est la solution et qu'on essaie
            de saisir le mot w. On calcule alors le score de w en évaluant le nombre
            de mots que les indices issues de la saisie de w élimineraient
        end note
        solver -> solver : evaluate_score_with_specific_combination\n(w, c1)

            solver -> dict : is_matching_word_specific_attempt(…)
            activate dict #FFBBBB
            solver <-- dict : true|false
            deactivate dict
        end loop
        deactivate solver
    end
    deactivate solver
end
